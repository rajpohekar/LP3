#include <iostream>
#include <vector>
using namespace std;

// Function to check if it's safe to place a queen
bool isSafe(vector<vector<int>> &board, int row, int col, int n) {
    // Check this row on left side
    for (int i = 0; i < col; i++)
        if (board[row][i])
            return false;

    // Check upper diagonal on left side
    for (int i = row, j = col; i >= 0 && j >= 0; i--, j--)
        if (board[i][j])
            return false;

    // Check lower diagonal on left side
    for (int i = row, j = col; i < n && j >= 0; i++, j--)
        if (board[i][j])
            return false;

    return true;
}

// Recursive function to solve N-Queen problem
void solveNQ(vector<vector<int>> &board, int col, int n, int &count) {
    // Base case: if all queens are placed
    if (col >= n) {
        count++;
        cout << "\nSolution " << count << ":\n";
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++)
                cout << board[i][j] << " ";
            cout << endl;
        }
        return;
    }

    // Try placing queen in all rows for this column
    for (int i = 0; i < n; i++) {
        if (isSafe(board, i, col, n)) {
            board[i][col] = 1;         // Place the queen
            solveNQ(board, col + 1, n, count); // Recur for next column
            board[i][col] = 0;         // Backtrack
        }
    }
}

void solveNQueens(int n) {
    vector<vector<int>> board(n, vector<int>(n, 0));

    // First Queen is fixed at (0,0)
    board[0][0] = 1;
    cout << "First Queen placed at (0,0)\n";

    int count = 0;
    solveNQ(board, 1, n, count);  // Start solving from column 1

    if (count == 0)
        cout << "\nNo solution exists when first Queen is fixed at (0,0)\n";
    else
        cout << "\nTotal solutions found: " << count << endl;
}

int main() {
    int n;
    cout << "Enter value of N: ";
    cin >> n;

    solveNQueens(n);
    return 0;
}
